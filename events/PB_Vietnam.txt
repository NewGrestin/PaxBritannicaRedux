country_event = {#Cambodia arise
	id = vietnam_associates.1
	title = vietnam_associates.1.t
	desc = vietnam_associates.1.d
	picture = GFX_report_PB_Paris_Event

	is_triggered_only = yes

	fire_only_once = yes
	
	hidden = yes
	
	immediate = {
		if = {
			limit = {
				AND = {
					741 = {
						is_controlled_by = BFE
					}
					VIN = {
						AND = {
							country_exists = yes
							has_war_with = BFE
						}
					}
				}
			}
			CAM = {
				transfer_state = 741
				declare_war_on = {
					target = BFE
					type = annex_everything
				}
			}
			741 = {
				create_unit = {
					division = "division_template = \"Infantry Division\" start_experience_factor = 0.6 start_equipment_factor = 1.0"
					owner = CAM
					count = 2
				}
			}
		}
		country_event = {
			id = vietnam_associates.2
			days = 22
		}
	}

	option = { #Standard option
		name = vietnam_associates.1.a

		ai_chance = {
			factor = 100
		}
	}
}
