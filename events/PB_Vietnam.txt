country_event = {#Cambodia arise
	id = vietnam_associates.1
	title = vietnam_associates.1.t
	desc = vietnam_associates.1.d
	picture = GFX_report_PB_Paris_Event

	is_triggered_only = yes

	fire_only_once = yes
	
	hidden = yes
	
	immediate = {
		VIN = {
			create_faction = VIN_united_front
		}
		if = {
			limit = {
				AND = {
					741 = {
						is_controlled_by = BFE
					}
					VIN = {
						AND = {
							has_war_with = BFE
						}
					}
				}
			}
			CAM = {
				transfer_state = 741
				load_oob = "CAM_1933"
				set_cosmetic_tag = CAM_Insurgency
			}
			VIN = { 
				add_to_faction = CAM
			}
			BFE = {
				declare_war_on = {
					target = CAM
					type = annex_everything
				}
			}
		}
		country_event = {
			id = vietnam_associates.2
			days = 22
		}
	}

	option = { #Standard option
		name = vietnam_associates.1.a

		ai_chance = {
			factor = 100
		}
	}
}
