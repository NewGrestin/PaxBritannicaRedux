			factor = 1
        	}

		## Franica is formed via the German post war tree

		if = { 
			limit = { 
				has_global_flag = PB_COE_FRANCIA_LARGE_OCT 
			}
			every_state = {
				limit = { check_variable = { culture_token = 10 } }
				add_core_of = OCT
			}
			release = OCT
		}

		if = { 
			limit = { 
				has_global_flag = PB_COE_FRANCIA_SMALL_OCT
			}
			OCT = { 
				add_state_core = 557 
				add_state_core = 31 
				add_state_core = 22 
			}
			release = OCT 
		}

		if = { 
			limit = { 
				has_global_flag = PB_COE_FRANCIA_NORMANDY 
			}
			15 = { 
				add_core_of = NRM 
			}
			NRM = { 
				transfer_state = 15 
				add_ideas = PB_Constitutionally_Enshrined_Neutrality
			}
			country_event = { id = COEGOV.1 days = 0 }
		}

		if = { 
			limit = { has_global_flag = PB_COE_ISRAEL_IN_FRANCE }

			ENG = { add_to_faction = ISR }
		}

				has_global_flag = PB_COE_FRANCIA_OCT_TWO_SEAS
