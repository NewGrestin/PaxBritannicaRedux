country_event = { #Fate of Venezuela
	id = annexation.3000
	title = annexation_venezuela
	desc = annexation_venezuela_desc
	picture = GFX_report_PB_Peace_Deals_Europe

	is_triggered_only = yes
	
	option = { #Release Venezuela Whole
		log = "[GetLogInfo]: event annex.3000 option annexation.3000.a"
		name = annexation.3000.a
		trigger = {
			is_subject = no
		}
		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				annexations_can_annex = yes
			}
		}

		if = {
			limit = { country_exists = VEN }
			custom_effect_tooltip = VEN_puppet_tooltip
		}
		else = {
			set_temp_variable = { tag_to_release = VEN }
			release_targeted_tag = yes
		}
		recheck_annexations = yes
	}

	option = { #Cancel
		log = "[GetLogInfo]: event annexation.3000 option annexation.cancel"
		name = annexation.cancel
		ai_chance = { base = 0 }
		recheck_annexations = yes
    }
}
