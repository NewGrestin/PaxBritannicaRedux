add_namespace = lon

country_event = {		#League of Nations Baltic Mandate
	id = lon.1
	title = lon.1.t
	desc = lon.1.d
	picture = GFX_report_PB_Military_Tribunal

	is_triggered_only = yes

	option = { #NOT APPLICABLE
		name = lon.1.a

		ai_chance = {
			factor = 80
		}

		hidden_effect = {

		RUS = {
			end_puppet = BAL
		}

		BAL = {
			remove_ideas = PB_BAL_Forest_Brothers
	        remove_ideas = PB_BAL_Denikin_Playground
			drop_cosmetic_tag = yes

			set_cosmetic_tag = BAL_LON_MANDATE

			set_politics = {
				ruling_party = ds
			}

			create_country_leader = {
				name = "Sir Richard McCreery"
				desc = "PB_NO_DESC"
				picture = "Portrait_BAL_McCreery.dds"
				expire = "1965.1.1"
				ideology = DS_VARIANT_06
			}
		}

		every_country = {
			limit = {
				has_war_with = BAL
			}
			white_peace = BAL
		}

		every_state = {
			limit = {
				is_core_of = BAL
				is_core_of = LIT
				NOT = {
					is_controlled_by = RUS
				}
			}
			BAL = {
				transfer_state = PREV
			}
		}

		ENG = {
			add_to_faction = BAL
		}

		country_event = { id = lon.2 } 

		}
	}
}
