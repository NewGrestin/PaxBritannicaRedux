country_event = { #Ottoman revolt against Russia
	id = exact_date_trigger.8
	hidden = yes
	fire_only_once = yes

	mean_time_to_happen = {
		days = 10
	}
	
	trigger = {
		original_tag = TUR
		TUR = {
			AND = {
				exists = yes
				is_subject_of = RUS
			}
		}
		RUS = {
			AND = {
				has_war_with = GER
				has_war_with = ENG
			}
		}
	}
	
	option = {
		hidden_effect = {

			RUS = {
				set_autonomy = {
                    target = TUR
                    autonomy_state = autonomy_free
                    end_wars = yes
                    end_civil_wars = yes
                }
			}
		if = {
			limit = {
				GER = {
					has_war_with = TUR
				}
			} 
			GER = {
				white_peace = TUR
			}
		}
		if = {
			limit = {
				ENG = {
					has_war_with = TUR
				}
			} 
			ENG = {
				white_peace = TUR
			}
		}
		TUR = {
			leave_faction = yes
			ENG = { add_to_faction = TUR }
			news_event = {
                id = PBNews.182
                days = 1
			}
			set_cosmetic_tag = TUR_REVOLT
		}
		ENG = {
			add_to_war = { 
                targeted_alliance = TUR 
                enemy = RUS 
                hostility_reason = asked_to_join
            }
		}
		RUS = {
			declare_war_on = {
                target = TUR
                type = annex_everything
            }
		}

		}
	}
}
