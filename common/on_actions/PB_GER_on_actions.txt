	#RhenishUnrestEffects
	on_monthly_GER = {
		effect = {
				if  = { 
					limit = {
						OR = {
						    check_variable = { Germany_Rhineland_Unrest > 50 }
							check_variable = { Germany_Rhineland_Unrest = 50 }
						}
						NOT = { 
							has_idea = PB_GER_Rhineland_Unrest_Critical
						}
					}
					hidden_effect = {
						remove_ideas = {
							PB_GER_Rhineland_Unrest_Moderate
							PB_GER_Rhineland_Unrest_Normal
						}
						add_ideas = PB_GER_Rhineland_Unrest_Critical
					}
				}
				else_if  = { 
					limit = {
						AND = {
							check_variable = { Germany_Rhineland_Unrest < 50 }
							check_variable = { Germany_Rhineland_Unrest > 25 }
						}
						NOT = { 
							has_idea = PB_GER_Rhineland_Unrest_Moderate

					}

						remove_ideas = {

							PB_GER_Rhineland_Unrest_Critical
						}
						add_ideas = PB_GER_Rhineland_Unrest_Moderate
					}
				}
				else_if = { 
					limit = {
						OR = {
						    check_variable = { Germany_Rhineland_Unrest < 25 }
							check_variable = { Germany_Rhineland_Unrest = 0 }
						}
					    NOT = { 
						    has_idea = PB_GER_Rhineland_Unrest_Normal
					    }
					}
					hidden_effect = {
						remove_ideas = {
							PB_GER_Rhineland_Unrest_Moderate
							PB_GER_Rhineland_Unrest_Critical
						}
						add_ideas = PB_GER_Rhineland_Unrest_Normal
					}
				}
				else_if = { 
					limit = {
						has_idea = PB_GER_German_Centralization
						OR = {
						    check_variable = { Germany_Rhineland_Unrest < 25 }
							check_variable = { Germany_Rhineland_Unrest = 0 }
						}
					    NOT = { 
						    has_idea = PB_GER_Rhineland_Unrest_Normal_Better
					    }

					}
				}
