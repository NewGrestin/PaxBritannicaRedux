scripted_map_modes = {
	social_development_map_mode = {
		top = {
			type = country
			color = {
				set_temp_variable = { alpha = 1.0 }
				set_temp_variable = { thickness = 0.5 }
				set_temp_variable = { selected_social_development_token = token:social_development_category_poverty }
				add_to_temp_variable = { selected_social_development_token = selected_social_development_map_mode_toggle }
				meta_trigger = {
					text = {
						set_temp_variable = { selected_social_development_num_of_levels = constant:PB_social_development_categories.[TOKEN].num_of_levels }
					}
					TOKEN = "[?selected_social_development_token.GetTokenKey]"
				}
				set_temp_variable = { selected_social_development_level = PB_social_development^selected_social_development_map_mode_toggle }
				divide_temp_variable = { selected_social_development_level = selected_social_development_num_of_levels }
				set_temp_variable = { red = 1 }
				subtract_from_temp_variable = { red = selected_social_development_level }
				set_temp_variable = { green = selected_social_development_level }
				set_temp_variable = { blue = 0 }
			}
			thickness = yes
			targets = { target_array = global.countries }
		}

		bottom = { type = none }

		far_text = none
		near_text = none

		update_daily = no
	}
}
