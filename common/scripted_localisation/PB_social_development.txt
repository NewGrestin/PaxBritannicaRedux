	name = get_poverty_rate_growth_type
	text = {
		trigger = {
			check_variable = {
				social_development_poverty_rate_monthly_change = 0
			}
		}
		localization_key = socdev_stagnant_growth
	}
	text = {
		trigger = {
			check_variable = {
				social_development_poverty_rate_monthly_change > 0
				social_development_poverty_rate_monthly_change < 1
			}
			NOT = {
				check_variable = {
					social_development_poverty_rate_monthly_change < 0
				}
			}
		}
		localization_key = socdev_slow_progress
	}
	text = {
		trigger = {
			check_variable = {
				social_development_poverty_rate_monthly_change > 0.99
				social_development_poverty_rate_monthly_change < 2.5
			}
			NOT = {
				check_variable = {
					social_development_poverty_rate_monthly_change < 0
				}
			}
		}
		localization_key = socdev_moderate_progress
	}
	text = {
		trigger = {
			check_variable = {
				social_development_poverty_rate_monthly_change > 2.49
			}
			NOT = {
				check_variable = {
					social_development_poverty_rate_monthly_change < 0
				}
			}
		}
		localization_key = socdev_rapid_progress
	}
	text = {
		trigger = {
			check_variable = {
				social_development_poverty_rate_monthly_change < 0
				social_development_poverty_rate_monthly_change > -1
			}
		}
		localization_key = socdev_slow_regress
	}
	text = {
		trigger = {
			check_variable = {
				social_development_poverty_rate_monthly_change < -0.99
				social_development_poverty_rate_monthly_change > -2.5
			}
		}
		localization_key = socdev_moderate_regress
	}
	text = {
		trigger = {
			check_variable = {
				social_development_poverty_rate_monthly_change < -2.49
			}
		}
		localization_key = socdev_rapid_regress
	}
}
