update_custom_state_pop_growth = {
	every_state = { 
		limit = { 
			OR = {
				has_state_flag = custom_state_pop_growth
				medical_center = 1
			}
		}
		set_temp_variable = { modified_pop_growth = state_population_k } 
		log = "Current state population of [THIS.Getname] in thousands is [?state_population_k]"
		log = "Current state population of [THIS.Getname] is [?state_population]"
		set_temp_variable = { custom_pop_growth = 1 }
		add_to_temp_variable = { custom_pop_growth = modifier@state_monthly_population } 
		multiply_temp_variable = { custom_pop_growth = 0.125 } 
		multiply_temp_variable = { modified_pop_growth = custom_pop_growth }
		multiply_temp_variable = { modified_pop_growth = 0.03 }
		log = "Custom growth of [THIS.Getname] is [?custom_pop_growth]"
		log = "The population of [THIS.Getname] will be changed by [?modified_pop_growth] people"
		add_manpower = var:modified_pop_growth
		log = "Current state population of [THIS.Getname] is [?state_population]"
	}
}
