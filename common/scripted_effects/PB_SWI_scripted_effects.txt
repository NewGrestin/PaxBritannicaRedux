SWI_start_civil_war = {
	set_global_flag = swiss_civil_war
	SWI = {
		add_named_threat = {
			threat = 3
			name = SWI_civil_war_threat
		}
		set_cosmetic_tag = SWI_PROTESTANT
		declare_war_on = {
			target = SWC
			type = civil_war
		}
	}
	every_core_state = {
		add_core_of = SWC
	}
	every_owned_state = {
		limit = { is_claimed_by = SWI }
		add_claim_by = SWC
	}

	add_to_temp_array = { SWI_protestant_states = 3.id }
	add_to_temp_array = { SWI_protestant_states = 151.id }
	add_to_temp_array = { SWI_protestant_states = 940.id }

	every_owned_state = {
		teleport_armies = { to_state_array = SWI_protestant_states }
	}

	every_owned_state = {
		limit = {
			is_controlled_by = ROOT
			OR = {
				state = 971 #Grisons
				state = 1403 #Ticino
				state = 1404 #Valais
			}
		}
		transfer_state_to = SWC
	}
	SWC = {
		inherit_technology = SWI
		set_capital = {
			state = 971 #Grisons
		}
	}
}
