retire_current_country_leader = {
	retire_country_leader = yes
	set_character_flag = PB_country_leader_retired_flag
}

kill_character = {
	custom_effect_tooltip = PB_kill_character_tt
	hidden_effect = {
		retire = yes
		set_character_flag_dead = yes
		if = {
			limit = { is_unit_leader = yes }
			remove_unit_leader_role = yes
		}
		remove_all_advisor_roles = yes
	}
}

set_character_flag_dead = {
	set_character_flag = PB_character_dead_flag
}

	remove_all_cabinet_roles = yes
	remove_military_advisor_roles = yes
