##### AI strategies to discriminate against horses #####

DOCTRINE_stop_producing_horses = {
	enable = {
		num_of_factories > 20
		date > 1935.1.1
	}
	abort = {
		always = no
	}

	ai_strategy = {
		type = role_ratio
		id = cavalry
		value = -100 # -100% cav
	}		
}
