default_build_nuclear_facility = {
	enable = {
		has_tech = atomic_research
		num_of_civilian_factories > 30
		num_of_military_factories > 30
		nuclear_facility < 1
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = build_building
		id = nuclear_facility
		value = 100
	}
}
