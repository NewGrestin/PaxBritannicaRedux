SWI_custom_volunteers_rule = {
	if = {
		limit = {
			original_tag = SWI
			FROM = { NOT = { has_country_flag = SWI_can_send_volunteers } }
			NOT = {
				any_country = {
					has_country_flag = SWI_can_send_volunteers
					has_war_with = FROM # Switzerland can't send volunteers to countries at war with countries that have this flag
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = SWI_VOLUNTEERS_NOT_INTEREST_TT
			hidden_trigger = {
				always = no
			}
		}
	}
	if = {
		limit = {
			original_tag = SWI
			any_country = {
				has_country_flag = SWI_can_send_volunteers
				has_war_with = FROM # Switzerland can't send volunteers to countries at war with countries that have this flag
			}
		}
		custom_trigger_tooltip = {
			tooltip = SWI_VOLUNTEERS_WAR_WITH_INTEREST_TT
			hidden_trigger = {
				always = no # Tooltip is shown when the condition is true
			}
		}
	}
}
