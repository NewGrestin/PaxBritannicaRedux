has_same_ideology = {
	custom_trigger_tooltip = {
		tooltip = has_same_ideology_tt
		OR = {
			AND = { #Neo Imperialist
				has_government = ni
				ROOT = { has_government = ni }
			}
			AND = { #Despotism
				has_government = ds
				ROOT = { has_government = ds }
			}
			AND = { #Reactionary
				has_government = rc
				ROOT = { has_government = rc }
			}
			AND = { #Anarcho-Liberal
				has_government = an
				ROOT = { has_government = an }
			}
			AND = { #Conservative
				has_government = cn
				ROOT = { has_government = cn }
			}
			AND = { #Liberal
				has_government = lb
				ROOT = { has_government = lb }
			}
			AND = { #Radical Egalitarian
				has_government = ss
				ROOT = { has_government = ss }
			}
			AND = { #Jacobin
				has_government = jc
				ROOT = { has_government = jc }
			}
		}
	}
}
has_far_right_government_group = {
	custom_trigger_tooltip = {
		tooltip = has_far_right_government_group_tt
		OR = {
			has_government = ni
			has_government = ds
			AND = {
				has_government = rc
				NOT = { has_reactionary_democratic_subideology = yes }
			}
			AND = {
				has_government = an
				NOT = { has_anlib_democratic_subideology = yes }
			}

		}
	}
}
