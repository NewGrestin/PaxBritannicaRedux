	is_core_of = ROOT
	is_owned_by = ROOT
}
mapmode_cores_is_core_not_owned = {
	AND = {
		is_core_of = ROOT
		NOT = { is_owned_by = ROOT }
	}

	is_claimed_by = ROOT
	is_owned_by = ROOT

	is_owned_by = ROOT

		ambition_area_@ROOT < 1
